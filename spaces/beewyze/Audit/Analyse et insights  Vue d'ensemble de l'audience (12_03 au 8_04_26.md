import io

def parse_ga_csv(file_path):
    with open(file_path, 'r', encoding='utf-8') as f:
        content = f.read()
    
    # Split content by empty lines or section headers
    # Each section seems to start with a date or a specific column name after comments
    sections = {}
    current_section_name = None
    current_lines = []
    
    lines = content.split('\n')
    for line in lines:
        line = line.strip()
        if not line:
            continue
        
        if line.startswith('#'):
            # It's a comment, maybe a new section name follows
            continue
        
        # Check if this line is a header (contains columns we recognize)
        if ',' in line:
            # If it's a known header pattern or we suspect a new table
            # Simple heuristic: if the first column is different from the previous table's first column
            # or if we haven't started a table yet.
            parts = line.split(',')
            # Let's use the full header line as a key for now
            if parts[0] in ["Nième jour", "Ville", "Page de destination + chaîne de requête"]:
                if current_section_name and current_lines:
                    sections[current_section_name] = pd.read_csv(io.StringIO('\n'.join(current_lines)))
                current_section_name = line
                current_lines = [line]
            else:
                current_lines.append(line)
    
    # Add the last section
    if current_section_name and current_lines:
        sections[current_section_name] = pd.read_csv(io.StringIO('\n'.join(current_lines)))
        
    return sections

import io
import pandas as pd

def extract_tables(file_path):
    with open(file_path, 'r', encoding='utf-8') as f:
        lines = f.readlines()
        
    tables = []
    current_table_lines = []
    
    for line in lines:
        line = line.strip()
        if not line or line.startswith('#'):
            if current_table_lines:
                # End of current table
                csv_data = "\n".join(current_table_lines)
                try:
                    df = pd.read_csv(io.StringIO(csv_data))
                    tables.append(df)
                except Exception as e:
                    print(f"Error parsing table: {e}")
                current_table_lines = []
            continue
        
        current_table_lines.append(line)
        
    if current_table_lines:
        csv_data = "\n".join(current_table_lines)
        try:
            df = pd.read_csv(io.StringIO(csv_data))
            tables.append(df)
        except Exception as e:
            print(f"Error parsing last table: {e}")
            
    return tables

all_tables = extract_tables(file_path)

for i, df in enumerate(all_tables):
    print(f"Table {i}: Columns: {df.columns.tolist()}, Rows: {len(df)}")
